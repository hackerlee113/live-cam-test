let tycgIframe; let currentSource = "primary"; function initTaoyuanCCTV() { tycgIframe = document.getElementById('tycg-cctv'); loadPrimary(); let loaded = false; tycgIframe.onload = () => { loaded = true; updateStatus(); console.log("✅ CCTV 已載入"); }; // 5 秒 fallback setTimeout(() => { if (!loaded && currentSource === "primary") { console.warn("⚠️ 主來源失敗，自動切備用"); loadBackup(); } }, 5000); } /* ======================== 載入來源 ======================== */ function loadPrimary() { currentSource = "primary"; tycgIframe.src = "https://cctvtraffic.tycg.gov.tw/camera150/?t=" + Date.now(); updateStatus(); } function loadBackup() { currentSource = "backup"; tycgIframe.src = "https://cctvtraffic.tycg.gov.tw/camera278/?t=" + Math.random(); updateStatus(); } /* ======================== 按鈕切換 ======================== */ function switchToPrimary() { loadPrimary(); } function switchToBackup() { loadBackup(); } /* ======================== UI 顯示 ======================== */ function updateStatus() { const status = document.getElementById('tycg-source-status'); if (currentSource === "primary") { status.innerText = "目前來源：主來源 (camera150)"; } else { status.innerText = "目前來源：備用來源 (camera278)"; } }
